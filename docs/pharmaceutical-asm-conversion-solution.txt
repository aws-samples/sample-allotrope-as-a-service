# Pharmaceutical Instrument Data to Allotrope Simple Model (ASM) Conversion Guide

## Project Overview

This guide outlines the implementation of a Python-based solution for converting pharmaceutical instrument data from native formats to the Allotrope Simple Model (ASM) format, with validation against ASM standards schemas.

---

## Architecture Overview

```
Native Instrument Data → Parser → Transformer → ASM Validator → ASM Output
                           ↓          ↓              ↓
                    Vendor SDKs   Mapping Logic   ASM Schemas
```

### 1. Base Parser Class

```python
# src/parsers/base_parser.py

from abc import ABC, abstractmethod
from pathlib import Path
from typing import Dict, Any
import logging

class BaseInstrumentParser(ABC):
    def __init__(self, file_path: Path):
        self.file_path = file_path
        self.logger = logging.getLogger(self.__class__.__name__)
        
    @abstractmethod
    def parse(self) -> Dict[str, Any]:
        pass
    
    @abstractmethod
    def validate_format(self) -> bool:
        pass
    
    def get_metadata(self) -> Dict[str, Any]:
        return {
            'file_name': self.file_path.name,
            'file_size': self.file_path.stat().st_size,
            'file_path': str(self.file_path)
        }
```

### 2. Instrument-Specific Parser Example

```python
# src/parsers/agilent_parser.py

import xml.etree.ElementTree as ET
from typing import Dict, Any
from .base_parser import BaseInstrumentParser

class AgilentParser(BaseInstrumentParser):
    
    def validate_format(self) -> bool:
        try:
            tree = ET.parse(self.file_path)
            root = tree.getroot()
            return root.tag == 'AgilentData'
        except Exception as e:
            self.logger.error(f"Format validation failed: {e}")
            return False
    
    def parse(self) -> Dict[str, Any]:
        if not self.validate_format():
            raise ValueError("Invalid Agilent file format")
        
        tree = ET.parse(self.file_path)
        root = tree.getroot()
        
        data = {
            'instrument_info': self._parse_instrument_info(root),
            'method_info': self._parse_method_info(root),
            'sample_info': self._parse_sample_info(root),
            'measurements': self._parse_measurements(root),
            'metadata': self.get_metadata()
        }
        
        return data
    
    def _parse_instrument_info(self, root) -> Dict[str, Any]:
        instrument = root.find('.//Instrument')
        return {
            'manufacturer': 'Agilent',
            'model': instrument.find('Model').text if instrument else None,
            'serial_number': instrument.find('SerialNumber').text if instrument else None,
            'software_version': instrument.find('SoftwareVersion').text if instrument else None
        }
    
    def _parse_method_info(self, root) -> Dict[str, Any]:
        method = root.find('.//Method')
        return {
            'method_name': method.find('Name').text if method else None,
            'method_version': method.find('Version').text if method else None,
            'parameters': self._extract_parameters(method)
        }
    
    def _parse_sample_info(self, root) -> Dict[str, Any]:
        sample = root.find('.//Sample')
        return {
            'sample_id': sample.find('ID').text if sample else None,
            'sample_name': sample.find('Name').text if sample else None,
            'injection_volume': sample.find('InjectionVolume').text if sample else None
        }
    
    def _parse_measurements(self, root) -> list:
        measurements = []
        for signal in root.findall('.//Signal'):
            measurements.append({
                'signal_type': signal.find('Type').text,
                'data_points': self._extract_data_points(signal)
            })
        return measurements
    
    def _extract_parameters(self, method_node) -> Dict[str, Any]:
        params = {}
        if method_node:
            for param in method_node.findall('.//Parameter'):
                params[param.get('name')] = param.text
        return params
    
    def _extract_data_points(self, signal_node) -> list:
        data = signal_node.find('Data')
        if data is not None:
            return [float(x) for x in data.text.split(',')]
        return []
```

### 3. ASM Transformer

```python
# src/transformers/asm_transformer.py

from typing import Dict, Any
from datetime import datetime
import uuid

class ASMTransformer:
    
    def __init__(self, asm_version: str = "1.0.0"):
        self.asm_version = asm_version
        
    def transform(self, parsed_data: Dict[str, Any], 
                  technique_type: str) -> Dict[str, Any]:
        
        asm_document = {
            "$asm.manifest": self._create_manifest(),
            "measurement aggregate document": {
                "measurement document": [
                    self._create_measurement_document(parsed_data, technique_type)
                ]
            }
        }
        
        return asm_document
    
    def _create_manifest(self) -> Dict[str, Any]:
        return {
            "identifier": str(uuid.uuid4()),
            "timestamp": datetime.utcnow().isoformat() + "Z",
            "asm version": self.asm_version,
            "file identifier": str(uuid.uuid4())
        }
    
    def _create_measurement_document(self, data: Dict[str, Any], 
                                    technique: str) -> Dict[str, Any]:
        
        doc = {
            "measurement identifier": str(uuid.uuid4()),
            "measurement time": datetime.utcnow().isoformat() + "Z",
            "analytical method identifier": data['method_info'].get('method_name'),
            "experimental data identifier": str(uuid.uuid4()),
            "device system document": self._create_device_document(data['instrument_info']),
            "sample document": self._create_sample_document(data['sample_info']),
            f"{technique.lower()} document": self._create_technique_document(
                data['measurements'], technique
            )
        }
        
        return doc
    
    def _create_device_document(self, instrument_info: Dict[str, Any]) -> Dict[str, Any]:
        return {
            "device identifier": instrument_info.get('serial_number'),
            "model number": instrument_info.get('model'),
            "equipment serial number": instrument_info.get('serial_number'),
            "brand name": instrument_info.get('manufacturer'),
            "product manufacturer": instrument_info.get('manufacturer'),
            "firmware version": instrument_info.get('software_version')
        }
    
    def _create_sample_document(self, sample_info: Dict[str, Any]) -> Dict[str, Any]:
        return {
            "sample identifier": sample_info.get('sample_id'),
            "description": sample_info.get('sample_name'),
            "sample volume setting": {
                "value": float(sample_info.get('injection_volume', 0)),
                "unit": "µL"
            }
        }
    
    def _create_technique_document(self, measurements: list, 
                                   technique: str) -> Dict[str, Any]:
        
        if technique == "HPLC":
            return self._create_hplc_document(measurements)
        elif technique == "LCMS":
            return self._create_lcms_document(measurements)
        
        return {}
    
    def _create_hplc_document(self, measurements: list) -> Dict[str, Any]:
        return {
            "chromatography aggregate document": {
                "chromatogram data cube": [
                    {
                        "cube-structure": {
                            "dimensions": [
                                {
                                    "concept": "retention time",
                                    "unit": "s"
                                }
                            ],
                            "measures": [
                                {
                                    "concept": "absorbance",
                                    "unit": "mAU"
                                }
                            ]
                        },
                        "data": self._format_measurement_data(measurements)
                    }
                ]
            }
        }
    
    def _create_lcms_document(self, measurements: list) -> Dict[str, Any]:
        # Implement LC-MS specific structure
        pass
    
    def _format_measurement_data(self, measurements: list) -> list:
        formatted_data = []
        for measurement in measurements:
            formatted_data.extend(measurement.get('data_points', []))
        return formatted_data
```

### 4. ASM Validator

```python
# src/validators/asm_validator.py

import jsonschema
from pathlib import Path
from typing import Dict, Any, List
import json
import logging

class ASMValidator:
    
    def __init__(self, schema_dir: Path):
        self.schema_dir = schema_dir
        self.logger = logging.getLogger(self.__class__.__name__)
        self.schemas = self._load_schemas()
        
    def _load_schemas(self) -> Dict[str, Any]:
        schemas = {}
        
        for schema_file in self.schema_dir.rglob("*.json"):
            with open(schema_file, 'r') as f:
                schema_name = schema_file.stem
                schemas[schema_name] = json.load(f)
                
        self.logger.info(f"Loaded {len(schemas)} ASM schemas")
        return schemas
    
    def validate(self, asm_document: Dict[str, Any], 
                 technique_type: str) -> tuple:
        
        errors = []
        
        try:
            if not self._validate_manifest(asm_document):
                errors.append("Invalid manifest structure")
            
            if not self._validate_measurement_document(asm_document):
                errors.append("Invalid measurement document structure")
            
            if not self._validate_technique_schema(asm_document, technique_type):
                errors.append(f"Invalid {technique_type} schema")
            
            if not self._validate_data_types(asm_document):
                errors.append("Invalid data types or units")
                
        except Exception as e:
            errors.append(f"Validation error: {str(e)}")
            self.logger.error(f"Validation failed: {e}")
        
        is_valid = len(errors) == 0
        return is_valid, errors
    
    def _validate_manifest(self, document: Dict[str, Any]) -> bool:
        manifest = document.get("$asm.manifest")
        if not manifest:
            return False
        
        required_fields = ["identifier", "timestamp", "asm version"]
        return all(field in manifest for field in required_fields)
    
    def _validate_measurement_document(self, document: Dict[str, Any]) -> bool:
        try:
            measurement_agg = document.get("measurement aggregate document")
            if not measurement_agg:
                return False
            
            measurements = measurement_agg.get("measurement document")
            if not measurements or not isinstance(measurements, list):
                return False
            
            return True
        except Exception:
            return False
    
    def _validate_technique_schema(self, document: Dict[str, Any], 
                                   technique: str) -> bool:
        schema_name = f"{technique.lower()}_schema"
        
        if schema_name not in self.schemas:
            self.logger.warning(f"Schema not found: {schema_name}")
            return True
        
        try:
            jsonschema.validate(
                instance=document,
                schema=self.schemas[schema_name]
            )
            return True
        except jsonschema.ValidationError as e:
            self.logger.error(f"Schema validation failed: {e}")
            return False
    
    def _validate_data_types(self, document: Dict[str, Any]) -> bool:
        return True
    
    def generate_validation_report(self, asm_document: Dict[str, Any],
                                   technique_type: str) -> Dict[str, Any]:
        is_valid, errors = self.validate(asm_document, technique_type)
        
        report = {
            "validation_timestamp": datetime.utcnow().isoformat() + "Z",
            "is_valid": is_valid,
            "technique_type": technique_type,
            "errors": errors,
            "warnings": [],
            "document_id": asm_document.get("$asm.manifest", {}).get("identifier")
        }
        
        return report
```

### 5. Main Conversion Pipeline

```python
# src/main.py

from pathlib import Path
from typing import Dict, Any
import json
import logging
from parsers.agilent_parser import AgilentParser
from transformers.asm_transformer import ASMTransformer
from validators.asm_validator import ASMValidator

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)

class InstrumentDataConverter:
    
    def __init__(self, schema_dir: Path):
        self.transformer = ASMTransformer()
        self.validator = ASMValidator(schema_dir)
        self.logger = logging.getLogger(self.__class__.__name__)
        
    def convert(self, input_file: Path, technique_type: str,
                output_file: Path = None) -> Dict[str, Any]:
        
        self.logger.info(f"Starting conversion: {input_file}")
        
        # Step 1: Parse native format
        parser = self._get_parser(input_file)
        parsed_data = parser.parse()
        self.logger.info("Parsing completed")
        
        # Step 2: Transform to ASM
        asm_document = self.transformer.transform(parsed_data, technique_type)
        self.logger.info("Transformation completed")
        
        # Step 3: Validate against ASM schemas
        is_valid, errors = self.validator.validate(asm_document, technique_type)
        
        if not is_valid:
            self.logger.error(f"Validation failed: {errors}")
            raise ValueError(f"ASM validation failed: {errors}")
        
        self.logger.info("Validation passed")
        
        # Step 4: Save output
        if output_file:
            self._save_output(asm_document, output_file)
            self.logger.info(f"Output saved: {output_file}")
        
        return asm_document
    
    def _get_parser(self, file_path: Path):
        if 'agilent' in file_path.name.lower():
            return AgilentParser(file_path)
        
        raise ValueError(f"No parser available for: {file_path}")
    
    def _save_output(self, asm_document: Dict[str, Any], output_file: Path):
        output_file.parent.mkdir(parents=True, exist_ok=True)
        
        with open(output_file, 'w') as f:
            json.dump(asm_document, f, indent=2)
    
    def batch_convert(self, input_dir: Path, technique_type: str,
                     output_dir: Path):
        
        input_files = list(input_dir.glob('*'))
        self.logger.info(f"Processing {len(input_files)} files")
        
        results = {
            'successful': [],
            'failed': []
        }
        
        for input_file in input_files:
            try:
                output_file = output_dir / f"{input_file.stem}_asm.json"
                self.convert(input_file, technique_type, output_file)
                results['successful'].append(str(input_file))
            except Exception as e:
                self.logger.error(f"Failed to convert {input_file}: {e}")
                results['failed'].append({
                    'file': str(input_file),
                    'error': str(e)
                })
        
        return results


# Example usage
if __name__ == "__main__":
    schema_dir = Path("schemas/asm")
    input_file = Path("data/input/sample_hplc_data.xml")
    output_file = Path("data/output/sample_hplc_asm.json")
    
    converter = InstrumentDataConverter(schema_dir)
    
    try:
        asm_document = converter.convert(
            input_file=input_file,
            technique_type="HPLC",
            output_file=output_file
        )
        print("Conversion successful!")
        print(f"Output: {output_file}")
    except Exception as e:
        print(f"Conversion failed: {e}")
```

def parallel_batch_convert(files: List[Path], technique: str, 
                          output_dir: Path, max_workers: int = 4):
    
    with ProcessPoolExecutor(max_workers=max_workers) as executor:
        futures = []
        for file in files:

## Troubleshooting

### Common Issues

**Issue**: Schema validation fails
- **Solution**: Ensure ASM schemas are up to date: `git pull` in schemas/asm directory

**Issue**: Parser fails to read file
- **Solution**: Verify file format matches expected vendor format; check file permissions

**Issue**: Missing required fields in output
- **Solution**: Review instrument_mappings.yaml and ensure all required fields are mapped
